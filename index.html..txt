function agregar(nombre, precio){

    if(inventario[nombre] <= 0){
        alert("Producto agotado");
        return;
    }

    inventario[nombre]--;

    listaProductos.push({nombre, precio});
    total += precio;

    let item = document.createElement("li");
    item.textContent = nombre + " - $" + precio;

    document.getElementById("lista").appendChild(item);
    document.getElementById("total").textContent = total;

    actualizarInventario();
}

function actualizarInventario(){

    let html = "";

    for(let producto in inventario){

        if(inventario[producto] == 0){
            html += `
            <tr style="background:#ffb3b3">
                <td>${producto}</td>
                <td>AGOTADO</td>
            </tr>`;
        } else {
            html += `
            <tr>
                <td>${producto}</td>
                <td>${inventario[producto]}</td>
            </tr>`;
        }
    }

    document.getElementById("tablaInventario").innerHTML = html;
}

function generarTicket(){

    let ticket = "Michi-Gan\n\n";

    listaProductos.forEach(p => {
        ticket += p.nombre + " - $" + p.precio + "\n";
    });

    ticket += "\n------------------";
    ticket += "\nTotal: $" + total;
    ticket += "\nPago: " + document.getElementById("pago").value;
    ticket += "\nGracias por tu compra";

    document.getElementById("ticket").textContent = ticket;
}
